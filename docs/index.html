Quick Installation

# Download and install
curl -L https://github.com/mohamadlounnas/ffan/releases/latest/download/ffan-v1.0.0-macos.zip -o ffan.zip
unzip ffan.zip && mv ffan.app /Applications/ && rm ffan.zip

# Enable fan control (requires sudo)
cd /Applications/ffan.app/Contents/Resources/tools/smc-helper
sudo ./install.sh
